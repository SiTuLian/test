canvas——直线

1.画一条直线

		<script type="text/javascript">

			var canvas = document.getElementById('canvas');//获取画布对象

			//设置宽高
			canvas.width = 500;
			canvas.height = 500;

			var draw = canvas.getContext('2d');//获取2d绘画环境，2d 即是 平面

			draw.beginPath();//创建路径，可以理解为拿起了一支画笔

			draw.lineWidth = 5;//设置画笔的粗细，如果不设置默认为1px,现在设置为5px
			draw.strokeStyle = "red";//设置画笔的颜色

			draw.moveTo(100,100);//起始点，将笔头点在 距离左边100像素，距离顶部100像素的位置
			draw.lineTo(100,300);//终止点，笔头从100,100 划至 100,300的位置

			draw.closePath();//关闭路径，把笔扔掉
			draw.stroke();//把画的路径显示出来

		</script>
您的浏览器不支持canvas，请换一个！

2.画多条直线

		<script type="text/javascript">

			var canvas = document.getElementById('canvas2');//获取画布对象

			//设置宽高
			canvas.width = 500;
			canvas.height = 500;

			var draw = canvas.getContext('2d');//获取2d绘画环境，2d 即是 平面

			draw.beginPath();//创建路径，可以理解为拿起了一支画笔
			draw.moveTo(100,100);//起始点，将笔头点在 距离左边100像素，距离顶部100像素的位置
			draw.lineTo(100,300);//终止点，笔头从100,100 划至 100,300的位置

			//想要画多条直线，只要重复moveTo(x,y) lineTo(x,y)即可
			draw.moveTo(300,100);
			draw.lineTo(100,300);

			draw.closePath();//关闭路径，把笔扔掉
			draw.stroke();//把画的路径显示出来

		</script>
